    ```rust
    /// GET /api/distillation/drafts — list draft (enabled=false) distilled skills.
    pub async fn list_distillation_drafts(State(state): State<Arc<AppState>>) -> impl IntoResponse {
        let reg = state.kernel.skill_registry.read().unwrap_or_else(|e| e.into_inner());
        let drafts: Vec<serde_json::Value> = reg
            .list_drafts()
            .iter()
            .map(|s| serde_json::json!({
                "name": s.manifest.skill.name,             // ADJUST to the real accessor (copy from list_skills)
                "description": s.manifest.skill.description, // ADJUST to the real accessor
                "enabled": s.enabled,
                "path": s.path.display().to_string(),
            }))
            .collect();
        let count = drafts.len();
        axum::Json(serde_json::json!({ "drafts": drafts, "count": count }))
    }

    /// POST /api/distillation/drafts/{name}/approve — flip a draft to enabled=true.
    pub async fn approve_distillation_draft(
        State(state): State<Arc<AppState>>,
        axum::extract::Path(name): axum::extract::Path<String>,
    ) -> impl IntoResponse {
        let result = {
            let mut reg = state.kernel.skill_registry.write().unwrap_or_else(|e| e.into_inner());
            reg.approve_draft_skill(&name)
        };
        match result {
            Ok(()) => (
                axum::http::StatusCode::OK,
                axum::Json(serde_json::json!({ "approved": name, "enabled": true })),
            ),
            Err(e) => (
                axum::http::StatusCode::BAD_REQUEST,
                axum::Json(serde_json::json!({ "error": e.to_string() })),
            ),
        }
    }
    ```
